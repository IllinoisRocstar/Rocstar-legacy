475 // MaximumTime, maxNumTimeSteps
476 bool reached_simulation_time( const Control_parameters &param) {
477  // if exceed max num of steps
478  if (param.cur_step>=param.maxNumTimeSteps) {
479  MAN_DEBUG(2, ("\nRocstar: reached_simulation_time returns TRUE with cur_step: %d\n", param.cur_step));
480  return true;
481  }
482 
483  int verb = param.controlVerb;
484 
485  // if exceed max wall time
486  double startTimeLoopTime;
487  if (param.myRank == 0) startTimeLoopTime = MPI_Wtime();
488  MPI_Bcast( &startTimeLoopTime, 1, MPI_DOUBLE, 0, param.communicator);
489  const double elaped_time = startTimeLoopTime - param.startTime;
490  if (elaped_time >= param.maxwalltime ) {
491  if (param.myRank == 0 && verb > 1)
492  printf("Rocstar: Quitting; elapsed wall clock time = %f\n", elaped_time);
493  return true;
494  }
495  else {
496  if (param.myRank == 0) {
497  MAN_DEBUG(2, ("\nRocstar: reached_simulation_time: elapsed wall time = %f \n", elaped_time));
498  }
499  }
500 
501  // if exceed MaximumTime the simulation time
502  if (param.current_time >= param.simue_time) { // MaximumTime
503  if (param.myRank == 0 && verb > 0)
504  printf("Rocstar: Quitting; CurrentTime = %e, MaximumTime = %e\n", param.current_time, param.simue_time);
505  return true;
506  }
507 
508  // if exceed the Maximum Number of dumps
509  if( (param.maxNumDumps > 0 ) && (param.current_dump >= param.maxNumDumps)){
510  if(param.myRank == 0 && verb > 0)
511  printf("Rocstar: Quitting; Reached the maximum number of dumps for this run. (%d)\n",param.current_dump);
512  return true;
513  }
514 
515  return false;
516 }

533 int check_for_interrupt(Coupling *coup,const Control_parameters &param)
534 {
535  int interrupt_code = 0;
536  std::string message("Interrupted by file");
537  // std::string therest;
538  if(param.myRank == 0) {
539  std::ifstream InterruptFile;
540  InterruptFile.open("RocstarInterrupt.txt");
541  if(InterruptFile){
542  std::cout << "Rocstar: Processing interrupt from file." << std::endl;
543  InterruptFile >> interrupt_code;
544  std::getline(InterruptFile,message);
545  InterruptFile.close();
546  unlink("RocstarInterrupt.txt");
547  std::ofstream InterruptOut;
548  InterruptOut.open("RocstarInterrupt.txt.processed");
549  InterruptOut << interrupt_code << " " << message << std::endl;
550  InterruptOut.close();
551  }
552  }
