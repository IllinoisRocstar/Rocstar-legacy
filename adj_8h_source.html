70 //A trivial grouped allocator
71 template <class T>
72 class allocPool {
73  enum {nAlloc=256}; //Number to allocate at once
74  T *buf;
75  int cur;
76  void fillBuffer(void) {
77  bufferCount++;
78  buf=(T*)malloc(sizeof(T)*nAlloc);
79  cur=0;
80  }
81 public:
82  int allocCount,bufferCount,freeCount;
83  allocPool() {
84  cur=nAlloc;
85  allocCount=bufferCount=freeCount=0;
86  }
87  void *alloc(void) {
88  allocCount++;
89  if (cur>=nAlloc) fillBuffer();
90  return &buf[cur++];
91  }
92  void free(void *) {
93  freeCount++;
94  /*otherwise ignored*/
95  }
96 };
