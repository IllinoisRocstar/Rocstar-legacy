246 void ComputeFluidLoad_ALE::run( double t, double dt, double alpha) {
247  MAN_DEBUG(3, ("Rocstar: ComputeFluidLoad_ALE::run() with t:%e dt:%e.\n", t, dt));
248  // ts = tf + ( mdot*Vs
249 
250  // part 1 (POST_UPDATE_FLUID in fluid_agent.f90)
251  if (traction_mode != NO_SHEER) {
252  COM_call_function( RocBlas::copy, &f_tf_hdl, &f_ts_hdl);
253 
254  // Compute tfmts = t_s-t_f (see developers guide).
255  // Here fb_mdot_tmp is used as buffer space.
256  COM_call_function( RocBlas::div, &fb_mdot_hdl, &fb_rhof_alp_hdl, &fb_mdot_tmp_hdl);
257 
258  if ( sagent->withALE) {
259  COM_call_function( RocBlas::sub, &fb_mdot_tmp_hdl, &b_rb_hdl, &fb_mdot_tmp_hdl);
260  }
261 
262  COM_call_function( RocBlas::mul, &fb_mdot_tmp_hdl, &fb_mdot_hdl, &fb_mdot_tmp_hdl);
263  COM_call_function( RocBlas::mul, &fb_nf_alp_hdl, &fb_mdot_tmp_hdl, &fb_ts_hdl);
264 
265  // Compute ts = tf - (tf-ts)
266  COM_call_function( RocBlas::sub, &fb_tf_hdl, &fb_ts_hdl, &fb_ts_hdl);
267  }
268  else { // ts is a scalar
269 
270  COM_call_function( RocBlas::copy, &f_pf_hdl, &f_ts_hdl);
271 
272  // Compute tsmpf = p_f-t_s (see developers guide).
273  // Here fb_mdot_tmp is used as buffer space.
274  COM_call_function( RocBlas::div, &fb_mdot_hdl, &fb_rhof_alp_hdl, &fb_mdot_tmp_hdl);
275 
276  if ( sagent->withALE) {
277  COM_call_function( RocBlas::sub, &fb_mdot_tmp_hdl, &b_rb_hdl, &fb_mdot_tmp_hdl);
278  }
279 
280  COM_call_function( RocBlas::mul, &fb_mdot_tmp_hdl, &fb_mdot_hdl, &fb_ts_hdl);
281 
282  // Compute ts = pf - (pf-ts)
283  COM_call_function( RocBlas::sub, &fb_pf_hdl, &fb_ts_hdl, &fb_ts_hdl);
284 
285  // Subtract from P_ambient if not zeros
286  double P_ambient = fagent->get_coupling()->get_rocmancontrol_param()->P_ambient;
287  if ( P_ambient != 0.0) {
288  COM_call_function( RocBlas::sub_scalar, &f_ts_hdl, &P_ambient, &f_ts_hdl);
289  }
290  }
