127 template < class FT >
128 PointS3<FT>::PointS3(const FT& x, const FT& y, const FT& z, const FT& w)
129 {
130  if (w != FT(1))
131  {
132  e0 = x/w;
133  e1 = y/w;
134  e2 = z/w;
135  }
136  else
137  {
138  e0 = x;
139  e1 = y;
140  e2 = z;
141  }
142 }
