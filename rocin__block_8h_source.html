131 struct GridInfo_CGNS {
133  inline GridInfo_CGNS(int* size, int cellDim, int ng)
134  : m_name(":st0:"), m_numElements(size[cellDim]),
135  m_numGhostElements(ng)
136  { m_name[3] += cellDim; m_size[0] = size[0]; m_size[1] = m_size[2] = 1;

139  } }
