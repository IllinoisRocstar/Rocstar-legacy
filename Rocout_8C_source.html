519 static bool is_option_value(const std::string& name, const std::string& val)
520 {
521  return ((name == "format"
522  && (val == "HDF" || val == "HDF4" || val == "HDF5" || val == "CGNS"))
523  || (name == "async"
524  && (val == "on" || val == "off"))
525  || (name == "mode"
526  && (val == "w" || val == "a"))
527  || (name == "localdir" /* && is_valid_path(val) */ )
528  || ((name == "rankwidth" || name == "pnidwidth") && is_whole(val))
529  || (name == "rankdir"
530  && (val == "on" || val == "off"))
531  || (name == "errorhandle"
532  && (val == "abort" || val == "ignore" || val == "warn"))
533  || (name == "ghosthandle"
534  && (val == "write" || val == "ignore")));

571 void Rocout::read_control_file( const char* filename)
572 {
573  // Attempt to open the control file.
574  std::ifstream fin(filename);
575  if (!fin.is_open()) {
576  ERROR_MSG("Rocout::read_control_file(): could not open file \""
577  << filename << "\".");
578  return;
579  }
580 
581  // Each line of the control file should be of the form "option_name=value",
582  // e.g. "format=HDF4". White space at the beginning, end, and around the
583  // '=' is ignored, as are empty lines and comments (lines that start with
584  // '#').
585  int lineNum = 0;
586  char buffer[256];
587  std::string line, name, val;
588  std::string::size_type eq;
589  while (!fin.eof()) {
590  ++lineNum;
591  // Read a line.
592  fin.getline(buffer, sizeof(buffer));
593  line = buffer;
594 
595  // Skip empty lines and comments.
596  if (line.empty() || line.find_first_not_of(" \t\n") == std::string::npos
597  || line[line.find_first_not_of(" \t\n")] == '#')
598  continue;
599 
600  // Find the '='. It's mandatory.
601  eq = line.find('=');
602  if (eq == std::string::npos) {
603  ERROR_MSG("Rocout::read_control_file(): option without value at line "
604  << lineNum << " of " << filename << '.');
605  continue;
606  }
607 
608  // Split the line into an option name and a value.
609  name = line.substr(0, eq);
610  val = line.substr(eq + 1);
611 
612  // Remove leading whitespace from the option name.
613  if (name.find_first_not_of(" \t\n") != 0)
614  name.erase(0, name.find_first_not_of(" \t\n"));
615 
616  // Check for an empty option name.
617  if (name.empty()) {
618  ERROR_MSG("Rocout::read_control_file(): missing option name at line "
619  << lineNum << " of " << filename << '.');
620  continue;
621  }
622 
623  // Remove trailing whitespace from the option name.
624  if (name.find_last_not_of(" \t\n") != name.length() - 1)
625  name.erase(name.find_last_not_of(" \t\n") + 1);
626 
627  // Make sure it is a valid option name.
628  if (!is_option_name(name)) {
629  ERROR_MSG("Rocout::read_control_file(): unknown option name \"" << name
630  << "\" at line " << lineNum << " of " << filename << '.');
631  continue;
632  }
633 
634  // Remove leading whitespace from the option value.
635  if (val.find_first_not_of(" \t\n") != 0)
636  val.erase(0, val.find_first_not_of(" \t\n"));
637 
638  // Check for an empty option value.
639  if (val.empty()) {
640  ERROR_MSG("Rocout::read_control_file(): option without value at line "
641  << lineNum << " of " << filename << '.');
642  continue;
643  }
644 
645  // Remove trailing whitespace from the option value.
646  if (val.find_last_not_of(" \t\n") != val.length() - 1)
647  val.erase(val.find_last_not_of(" \t\n") + 1);
648 
649  // Make sure it is a valid value for the named option.
650  if (!is_option_value(name, val)) {
651  ERROR_MSG("Rocout::read_control_file(): invalid value \"" << val
652  << "\" for option \"" << name << "\" at line " << lineNum
653  << " of " << filename << '.');
654  continue;
655  }
656 
657  set_option(name.c_str(), val.c_str());
658  }
659 }

661 void* Rocout::write_attr_internal(void* attrInfo)
662 {
663  WriteAttrInfo* ai = static_cast<WriteAttrInfo*>(attrInfo);
664  const Attribute* attr = ai->m_attr;
665  Window* tempWin = NULL;
666 
667  if (ai->m_cloned) {
668 #ifdef USE_PTHREADS
669  _writesem.Wait();
670 #endif // USE_PTHREADS
671  tempWin = new Window(attr->window()->name(),
672  attr->window()->get_communicator());
673  attr = tempWin->inherit(const_cast<Attribute*>(attr), attr->name(),
674  Pane::INHERIT_CLONE, true, NULL, 0);
675  }
676 
677  int flag = 0; MPI_Initialized(&flag);
678 
679  // Obtain process rank
680  int rank;
681  if ( flag) {
682  MPI_Comm wcomm;
683  const MPI_Comm* pComm;
684 
685  if ( ai->m_pComm)
686  pComm = ai->m_pComm;
687  else {
688  wcomm = attr->window()->get_communicator();
689  pComm = &wcomm;
690  }
691 
692  if (*pComm != MPI_COMM_NULL)
693  MPI_Comm_rank(*pComm, &rank);
694  else
695  rank = 0;
696  }
697  else
698  rank = 0;
699 
700  int append = ai->m_append;
701  if (append < 0) {
702  if (ai->m_rout->_options["mode"] == "w")
703  append = 0;
704  else
705  append = 1;
706  }
707 
708  std::vector<int> paneIds;
709  COM_get_panes(attr->window()->name().c_str(), paneIds);
710 
711  std::vector<int>::iterator begin = paneIds.begin(), end = paneIds.end(), p;
712  if (ai->m_pPaneId != NULL && *(ai->m_pPaneId) > 0) {

764 std::string Rocout::get_fname(const std::string& prefix,
765  int rank /* = -1 */,
766  int paneId /* = 0 */,
767  bool check /* = false */)
768 {
769  // Modify the prefix using the "localdir" option.
770  std::string pre(_options["localdir"]);
771  if (!pre.empty()) {
772  // Make sure there's exactly one '/' between the localdir and given prefix.
773  if (prefix[0] == '/') {
774  if (pre[pre.length()-1] == '/')
775  pre.erase(pre.length() - 1);
776  } else {
777  if (pre[pre.length()-1] != '/')
778  pre += '/';
779  }
780  }
781  pre += prefix;
782 
783  if (_options["rankdir"] == "on") { // write output file in <rank> dir
784  std::ostringstream rank_prefix;
785  rank_prefix << "/" << rank;
786  std::string::size_type s = pre.find_last_of('/');
787  if (s != std::string::npos)
788  pre.insert(s, rank_prefix.str());
789  }
790 
791  // Make sure the directory exists. Ignore any errors except for the last.
792  int result = 0;
793  std::string::size_type s = pre.find('/', 1);
794  while (s != std::string::npos) {
795  result = mkdir(pre.substr(0, s).c_str(), 0755);
796  s = pre.find('/', s + 1);
797  }
798  if (result < 0 && errno != EEXIST) {
799  ERROR_MSG("Rocout::write_attribute(): could not create directory '"
800  << pre.substr(0, pre.rfind('/') + 1) << "'.");
801  }
802 
803  if ( pre.find(".hdf") == pre.size()-4) {
804  _options["format"] = "HDF4";
805  return pre;
806  }
807  else if (pre.find(".cgns") == pre.size()-5) {
808  _options["format"] = "CGNS";
809  return pre;
810  }
811 
812  if (rank < 0) {
813  int flag = 0;
814  MPI_Initialized(&flag);
815  if (flag)
816  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
817  else
818  rank = 0;
819  }
820 
821  std::string name;
822 
823  if (!pre.empty()) {
824  int rw, pw;
825  {
826  std::istringstream sin(_options["rankwidth"]);
827  sin >> rw;
828  }
829  {
830  std::istringstream sin(_options["pnidwidth"]);
831  sin >> pw;
832  }
833 
834  std::ostringstream sout;
835  sout << pre;
836  if (rw > 0)
837  sout << std::setw(rw) << std::setfill('0') << rank;
838  if (pw > 0 && paneId > 0) {
839  if (rw > 0)
840  sout << _options["separator"];
841  sout << std::setw(pw) << std::setfill('0') << paneId;
842  }
843 
844  const std::string fmt = _options["format"];
845  if ( fmt == "HDF" || fmt == "HDF4")
846  sout << ".hdf";
847  else if (fmt == "HDF5")
848  sout << ".hdf5";
849  else if (fmt == "CGNS")
850  sout << ".cgns";
851 
852  name = sout.str();
853 
854  if (check) {
855  std::FILE* f;
856  if ((f = std::fopen(name.c_str(), "r")) != NULL)
857  std::fclose(f);
858  else if ((f = std::fopen(name.c_str(), "w")) != NULL) {
859  std::fclose(f);
860  std::remove(name.c_str());
861  } else {
862  // Change the directory to current directory
863  std::string::size_type n = name.rfind('/');
864  if (n != std::string::npos)
865  name.erase(0, n + 1);
866  }
867  }
868  }
869 
870  return name;
871 }
