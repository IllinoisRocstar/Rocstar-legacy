669  CALL rflu_destroygrid(pregion)
670  END DO ! iReg
671 
672 ! ==============================================================================
673 ! Destroy partition mapping
674 ! ==============================================================================
675 
676  CALL rflu_rnmb_destroysc2rmap(pregionserial)
677  END IF ! global%nRegionsLocal
678 
679 ! ******************************************************************************
680 ! Write restart info file
681 ! ******************************************************************************
682 
683  CALL rflu_writerestartinfo(global)
684 
685 ! ******************************************************************************
686 ! Deallocate memory
687 ! ******************************************************************************
688 
689  IF ( rflu_sype_havesypepatches(pregionserial) .EQV. .true. ) THEN
690  CALL rflu_sype_destroyp2vclist(pregionserial)
691  END IF ! RFLU_SYPE_HaveSyPePatches
692 
693  IF ( rflu_decideneedgridspeeds(pregionserial) .EQV. .true. ) THEN
694  CALL rflu_deallocatememorygspeeds(pregionserial)
695  END IF ! RFLU_DecideNeedGridSpeeds
696 
697  IF ( global%solverType == solv_implicit_nk ) THEN
698  CALL rflu_col_destroycoloring(pregionserial)
699  END IF ! global%solverType
700 
701  CALL rflu_destroyfacelist(pregionserial)
702  CALL rflu_destroybvertexlists(pregionserial)
703  CALL rflu_destroycellmapping(pregionserial)
704 
705 ! ******************************************************************************
706 ! Print info about warnings
707 ! ******************************************************************************
708 
709  CALL rflu_printwarninfo(global)
710 
711 ! ******************************************************************************
712 ! End
713 ! ******************************************************************************
714 
715  CALL deregisterfunction(global)
716 
717 END SUBROUTINE rflupart
