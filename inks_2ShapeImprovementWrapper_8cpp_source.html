51 ShapeImprovementWrapper::ShapeImprovementWrapper(MsqError& err,
52  double cpu_time,
53  double grad_norm)
54  : inverseMeanRatio(0),
55  objFunc(0),
56  feasNewt(0),
57  mQA(0),
58  termOuter(0),
59  termInner(0)
60 {
61 
62  //arbitrarily chosen variables
63  untBeta=1.e-8;
64  successiveEps=1.e-4;
65 
66  inverseMeanRatio = new IdealWeightInverseMeanRatio(err); MSQ_ERRRTN(err);
67  inverseMeanRatio->set_gradient_type(QualityMetric::ANALYTICAL_GRADIENT);
68  inverseMeanRatio->set_hessian_type(QualityMetric::ANALYTICAL_HESSIAN);
69  inverseMeanRatio->set_averaging_method(QualityMetric::LINEAR,err); MSQ_ERRRTN(err);
70  // creates the l_2 squared objective function
71  objFunc = new LPtoPTemplate(inverseMeanRatio, 2, err); MSQ_ERRRTN(err);
72  objFunc->set_gradient_type(ObjectiveFunction::ANALYTICAL_GRADIENT);
73  //creates a FeasibleNewtone improver
74  feasNewt = new FeasibleNewton(objFunc);
75  feasNewt->set_patch_type(PatchData::GLOBAL_PATCH, err,1 ,1); MSQ_ERRRTN(err);
76 
77  mQA = new QualityAssessor(inverseMeanRatio,QualityAssessor::MAXIMUM, err); MSQ_ERRRTN(err);
78  mQA->disable_printing_results();
79  //**************Set stopping criterion*e***************
80  termInner = new TerminationCriterion();
81  termOuter = new TerminationCriterion();
82  //termInner->add_criterion_type_with_double(TerminationCriterion::GRADIENT_L2_NORM_ABSOLUTE,grad_norm,err); MSQ_ERRRTN(err);
83  //termInner->add_criterion_type_with_double(TerminationCriterion::SUCCESSIVE_IMPROVEMENTS_RELATIVE,successiveEps,err); MSQ_ERRRTN(err);
84  termInner->add_criterion_type_with_int(TerminationCriterion::NUMBER_OF_ITERATES,1,err); MSQ_ERRRTN(err);
85  termOuter->add_criterion_type_with_int(TerminationCriterion::NUMBER_OF_ITERATES,1,err); MSQ_ERRRTN(err);
86  feasNewt->add_culling_method(PatchData::NO_BOUNDARY_VTX);
87  feasNewt->set_inner_termination_criterion(termInner);
88  feasNewt->set_outer_termination_criterion(termOuter);
89 
90 }
