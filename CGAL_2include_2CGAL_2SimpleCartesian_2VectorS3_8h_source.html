123 template < class FT >
124 VectorS3<FT>::VectorS3(const FT& x, const FT& y, const FT& z, const FT& w)
125 {
126  if (w != FT(1))
127  {
128  e0 = x/w;
129  e1 = y/w;
130  e2 = z/w;
131  }
132  else
133  {
134  e0 = x;
135  e1 = y;
136  e2 = z;
137  }
138 }
