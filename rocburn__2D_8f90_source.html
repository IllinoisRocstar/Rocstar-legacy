338  blk%rhoc_old = blk%rhoc
339  blk%Tg_old = blk%Tg

346  blk%pres(ic), zero, zero, blk%coor(1:3,ic), &

349  END IF !APN

351  END DO ! ib

360 
361  END IF !initial_time == 0.0
