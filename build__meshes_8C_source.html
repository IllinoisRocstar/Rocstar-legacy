153 void init_crds1_tet_unstr_2(double* crds){
154  for(int z = 0; z < 3; ++z){
155  for(int y = 0; y < 3; ++y){
156  for(int x = 0; x < 3; ++x){
157  int index = z*9 + y*3 + x;
158  crds[index*3]=x;
159  crds[index*3+1]=2-y;
160  crds[index*3+2]=2-z;
161  // if((index == 13)||(index == 14))
162  if(index == 14)
163  crds[index*3] += .33;
164  }
165  }
166  }
167 }
