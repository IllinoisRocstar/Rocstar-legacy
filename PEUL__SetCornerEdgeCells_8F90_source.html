24 !
25 ! Purpose: set values in edge and corner cells of a region for smoke.
26 !
27 ! Description: this is for the case if the cells are not located
28 ! within another region.
29 !
30 ! Input: region = region data.
31 !
32 ! Output: region%levels%peul%cv = cv variables in dummy cells.
33 !
34 ! Notes: values in ALL edge and corner cells are averaged from neighboring
35 ! cells. Hence, this routine should be called first, before values
36 ! from adjacent regions are copied.
37 !
38 !******************************************************************************
39 !
40 ! $Id: PEUL_SetCornerEdgeCells.F90,v 1.3 2008/12/06 08:44:39 mtcampbe Exp $
41 !
42 ! Copyright: (c) 2004 by the University of Illinois
43 !
44 !******************************************************************************

46 SUBROUTINE peul_setcorneredgecells( region )
47 
48  USE moddatatypes

125  ENDDO
126  ENDDO
127  ENDDO
128

130 
131  DO j=jpcbeg,jpcend

133  DO k=kpcend+1,kdcend
134  ijkd = indijk(i ,j,k ,icoff,ijcoff)
135  ijkc1 = indijk(i+1,j,k ,icoff,ijcoff)
136  ijkc2 = indijk(i ,j,k-1,icoff,ijcoff)

139  DO k=kpcbeg-1,kdcbeg,-1
140  ijkd = indijk(i ,j,k ,icoff,ijcoff)
141  ijkc1 = indijk(i+1,j,k ,icoff,ijcoff)
142  ijkc2 = indijk(i ,j,k+1,icoff,ijcoff)
143  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
144  ENDDO
145  ENDDO
146 
147  DO i=ipcend+1,idcend ! edges 6, 8
148  DO k=kpcend+1,kdcend
149  ijkd = indijk(i ,j,k ,icoff,ijcoff)
150  ijkc1 = indijk(i-1,j,k ,icoff,ijcoff)
151  ijkc2 = indijk(i ,j,k-1,icoff,ijcoff)
152  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
153  ENDDO
154  DO k=kpcbeg-1,kdcbeg,-1
155  ijkd = indijk(i ,j,k ,icoff,ijcoff)
156  ijkc1 = indijk(i-1,j,k ,icoff,ijcoff)
157  ijkc2 = indijk(i ,j,k+1,icoff,ijcoff)
158  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
159  ENDDO
160  ENDDO
161  ENDDO
162 
163 ! edges 1, 3, 5, 7 ------------------------------------------------------------
164 
165  DO k=kdcbeg,kdcend
166  DO i=ipcbeg-1,idcbeg,-1 ! edges 1, 3
167  DO j=jpcbeg-1,jdcbeg,-1
168  ijkd = indijk(i ,j ,k,icoff,ijcoff)
169  ijkc1 = indijk(i+1,j ,k,icoff,ijcoff)
170  ijkc2 = indijk(i ,j+1,k,icoff,ijcoff)
171  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
172  ENDDO
173  DO j=jpcend+1,jdcend
174  ijkd = indijk(i ,j ,k,icoff,ijcoff)
175  ijkc1 = indijk(i+1,j ,k,icoff,ijcoff)
176  ijkc2 = indijk(i ,j-1,k,icoff,ijcoff)
177  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
178  ENDDO
179  ENDDO
180 
181  DO i=ipcend+1,idcend ! edges 5, 7
182  DO j=jpcbeg-1,jdcbeg,-1
183  ijkd = indijk(i ,j ,k,icoff,ijcoff)
184  ijkc1 = indijk(i-1,j ,k,icoff,ijcoff)
185  ijkc2 = indijk(i ,j+1,k,icoff,ijcoff)
186  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
187  ENDDO
188  DO j=jpcend+1,jdcend

190  ijkc1 = indijk(i-1,j ,k,icoff,ijcoff)
191  ijkc2 = indijk(i ,j-1,k,icoff,ijcoff)
192  cv(:,ijkd) = 0.5_rfreal*(cv(:,ijkc1) + cv(:,ijkc2))
193  ENDDO
194  ENDDO
195  ENDDO
