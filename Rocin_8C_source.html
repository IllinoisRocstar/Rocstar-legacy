288 static int32 FindFirstGeometryDataset(int32 sd_id, const std::string& geomFile,
289  const std::string& matName,
290  const std::string& label, int32 hIndex,
291  int32* index)
292 {
293  int32 geom_id;
294  if (geomFile.empty()) {
295  // We're not using a separate geometry file, so just return the dataset
296  // immediately after the block header.
297  geom_id = sd_id;
298  *index = hIndex + 1;
299  return geom_id;
300  }
301 
302  // Open the geometry file.
303  HDF4_CHECK_RET(geom_id = HDF4::SDstart, (geomFile.c_str(), DFACC_READ),
304  return FAIL);
305 
306  // Get the number of datasets.
307  int32 dsCount = MAX_NC_VARS, nAttrs, status;
308  HDF4_CHECK(HDF4::SDfileinfo, (geom_id, &dsCount, &nAttrs));
309 
310  int32 sds_id, rank, size[5], dType;
311  char name[MAX_NC_NAME];
312  // We're looking for the dataset for this material and section.
313  std::string target = matName + '|' + label;
314 
315  // Scan through the datasets of the geometry file.
316  for (*index=0; *index<dsCount; ++(*index)) {
317  HDF4_CHECK_RET(sds_id = HDF4::Select,
318  (geom_id, *index, name, &rank, size, &dType, &nAttrs, dsCount),
319  HDF4::SDend(geom_id); return FAIL);
320 
321  HDF4_CHECK_RET(status = HDF4::SDgetdatastrs,
322  (sds_id, NULL, NULL, NULL, name, MAX_NC_NAME),
323  HDF4::SDendaccess(sds_id); continue);
324  HDF4::SDendaccess(sds_id);
325 
326  // Check if this is the right dataset.
327  if (strcasecmp(target.c_str(), name) == 0) { // We've found it!
328  return geom_id;
329  }
330  }
331 
332  // No luck.
333  HDF4::SDend(geom_id);
334  return FAIL;
335 }

343 static void scan_files_HDF4( int pathc, char* pathv[],
344  BlockMM_HDF4& blocks, std::string& time,
345  std::map<int32, COM_Type>& HDF2COM)
346 {
347  int32 sd_id, geom_id, sds_id, nDataSets, nAttrs, index, index3;
348  int32 rank, dType, start[1];
349  int32 size[5] = { 0, 0, 0, 0, 0 };
350  int32 geomIdx[3];
351  char name[MAX_NC_NAME], label[MAX_NC_NAME];
352  char timeLevel[MAX_NC_NAME], units[MAX_NC_NAME];
353  char matName[MAX_NC_NAME], varName[MAX_NC_NAME], varType[MAX_NC_NAME];
354  int xyz;
355  int numPoints=0, ghostPoints, gc, gridType=0;
356  std::vector<int> ghostCells;
357  char grid[32], ghost[32], location;
358  std::string geomFile;
359  char buffer[1024];
360  Block_HDF4* block;
361  int i;
362 
363  blocks.clear();
364 
365  // Iterate through each file.
366  for (i=0; i<pathc; ++i) {
367  // Make sure this is an HDF file.
368  if (strcmp(&pathv[i][strlen(pathv[i])-4], ".hdf") != 0
369  && strcmp(&pathv[i][strlen(pathv[i])-5], ".hdf4") != 0)
370  continue;
371 
372  HDF4_CHECK_RET(sd_id = HDF4::SDstart, (pathv[i], DFACC_READ), continue);
373  AutoCloser<int32, intn> auto0(sd_id, HDF4::SDend);
374 
375  // Determine the number of datasets.
376  HDF4_CHECK_RET(HDF4::SDfileinfo, (sd_id, &nDataSets, &nAttrs),
377  continue);
378 
379  // Iterate through the datasets.
380  for (index=0; index<nDataSets; ++index) {
381  // Select each non-"fakeDim" dataset in the file.
382  HDF4_CHECK_RET(sds_id = HDF4::Select,
383  (sd_id, index, name, &rank, size, &dType, &nAttrs, nDataSets),
384  continue);
385 
386  {
387  AutoCloser<int32, intn> auto1(sds_id, HDF4::SDendaccess);
388 
389  // Get the paneId ('label'), time level ('timeLevel'),
390  // dataset type ('varName'), and window name ('matName').
391  HDF4_CHECK_RET(HDF4::SDgetdatastrs,
392  (sds_id, label, timeLevel, varName, matName, MAX_NC_NAME),
393  continue);
394 
395  // If this isn't a block header, then it isn't interesting.
396  if (strncasecmp(varName, "block header", 12) != 0) {
397  continue;
398  } else {
399  // Otherwise, set the time level if not yet set, and skip
400  // the blocks that have different time levels.
401  if (time.empty())
402  time = timeLevel;
403  else if (time != timeLevel)
404  continue;
405  }
406 
407  // Block header datasets should have only one dimension.
408  if (rank != 1) {
409  std::cerr << "Rocstar: Warning: Dataset " << index << " in file " << pathv[i]
410  << " is marked as a block header, but its rank is not 1 "
411  << "(rank == " << rank << ')' << std::endl;
412  continue;
413  }
414 
415  ghostCells.clear();
416  geomFile.erase();
417  if (dType == DFNT_CHAR8) {
418  // This data file is written for version 1.1. Good!
419  start[0] = 0;
420  std::vector<char8> data(size[0]);
421  HDF4_CHECK_RET(HDF4::SDreaddata,
422  (sds_id, start, NULL, size, (VOIDP)&data[0]),
423  continue);
424 
425  // Parse the data into grid type, ghost info, and external
426  // geometry file.
427  std::istringstream sin(&data[0]);
428  sin.getline(grid, sizeof(grid), '|');
429  sin.getline(ghost, sizeof(ghost), '|');
430  buffer[0] = '\0';
431  sin >> buffer;
432 
433  // The first token is the grid type.
434  if (strcasecmp(grid, "structured") == 0 || strcmp(grid, "2") == 0) {
435  gridType = 2;
436  } else if (strcasecmp(grid, "unstructured") == 0
437  || strcmp(grid, "3") == 0) {
438  gridType = 3;
439  } else if (strcasecmp(grid, "discontinuous") == 0
440  || strcmp(grid, "4") == 0) {
441  gridType = 4;
442  } else if (strcasecmp(grid, "polydata") == 0
443  || strcmp(grid, "5") == 0) {
444  gridType = 5;
445  } else {
446  std::cerr << "Rocstar: Warning: The block header (dataset " << index
447  << " in file " << pathv[i] << ") does not have a "
448  << "valid grid type (grid = " << grid << ')'
449  << std::endl;
450  }
451 
452  // The second token is the number of ghost points and cells. If
453  // there is a single number, then this is (probably) the number of
454  // layers of ghost cells on a structured grid. Otherwise, it's the
455  // number of ghost points followed by the number of ghost cells of
456  // each connectivity table of an unstructured grid.
457  // WARNING: When using external grid files, the ghost data will
458  // probably not be accurate!
459 
460  ghostPoints = 0;
487  gc = atoi(ghost);
488  char* c = strchr(ghost, ',');
489  if (c != NULL) {
490  ++c;
491  ghostPoints = gc;
492  gc = atoi(c);
493  c = strchr(c, ',');
494  }
495  if (gc < 0) std::cerr << "Rocstar: Warning: Bad num ghosts " << gc << " from string: " << ghost << std::endl;
496  ghostCells.push_back(gc);
497  // Check if there are more data.
498  while (c != NULL) {
499  ++c;
500  gc = atoi(c);
501  c = strchr(c, ',');
502  if (gc < 0) std::cerr << "Rocstar: Warning: Bad ghosts " << gc << std::endl;
503  ghostCells.push_back(gc);
504  }
505 
506  // The third token (if present) is an external geometry file.
507  if (buffer[0] != '\0') {
508  if (buffer[0] == '/') {
509  geomFile = buffer;
510  } else {
511  // Normalize the filename.
512  geomFile = pathv[i];
513  std::string::size_type pos = geomFile.find_last_of('/');
514  if (pos != std::string::npos) {
515  geomFile.erase(pos + 1);
516  geomFile += buffer;
517 
518  // If normalized filename does not exist, then
519  // try to find the mesh file with the given name.
520  struct stat sb;
521  if (stat(geomFile.c_str(), &sb) != 0)
522  geomFile = buffer;
523  }
524  else
525  geomFile = buffer;
526  }
527 
528  // Make sure the file exists.
529  struct stat sb;
530  if (stat(geomFile.c_str(), &sb) != 0) {
531  std::cerr << "Rocstar: Warning: The block header (dataset " << index
532  << " in file " << pathv[i] << ") has an invalid "
533  << "filename for the external geometry data. The "
534  << "file does not exist: " << geomFile << std::endl;
535  continue;
536  }
537  }
538  } else {
539  std::cerr << "Rocstar: Warning: Dataset " << index << " in file " << pathv[i]
540  << " is marked as a block header, but its datatype is "
541  << "not char8." << std::endl;
542  continue;
543  }
544  }
545 
546  geom_id = FindFirstGeometryDataset(sd_id, geomFile, matName, label,
547  index, &index3);
548  if (geom_id == FAIL) {
549  std::cerr << "Rocstar: Warning: Could not open geometry dataset for section "
550  << label << " of material " << matName << std::endl;
551  continue;
552  }
553 
554  // Collect data on the nodal coordinate datasets.
555  for (xyz=0; xyz<6; xyz+=2,++index3) {
556  HDF4_CHECK_RET(sds_id = HDF4::Select,
557  (geom_id, index3, name, &rank, size, &dType, &nAttrs),
558  continue);
559  AutoCloser<int32, intn> auto2(sds_id, HDF4::SDendaccess);
560 
561  geomIdx[xyz/2] = index3;
562 
563  if (xyz == 0) {
564  units[0] = '\0';
565  if ((gridType == 2 && rank != 3) || (gridType != 2 && rank != 1)) {
566  numPoints = size[0] = size[1] = size[2] = 0;
567  geomIdx[0] = geomIdx[1] = geomIdx[2] = FAIL;
568  break;
569  }
570 
571  HDF4_CHECK(HDF4::SDgetdatastrs,
572  (sds_id, name, units, NULL, varType, MAX_NC_NAME));
573 
574  if (gridType == 2) {
575  numPoints = size[0] * size[1] * size[2];
576  } else {
577  char* pos = strchr(name, '|');
578 
579  if (pos && *(pos+2)=='0')
580  numPoints = ghostPoints = 0; // Empty pane.
581  else {
582  numPoints = size[0];
583  // Get a trustworthy count of the number of ghost nodes.
584  std::istringstream in(pos+3);
585  in >> ghostPoints;
586  }
587  }
588  }
589  }
590 
591  // Determine the pane id. Skip any non-numeric prefix such as "block".
592  std::string lbl(label);
593  std::string::size_type pos = lbl.find_first_of("0123456789");
594  if (pos != std::string::npos)
595  lbl.erase(0, pos);
596 
597  std::istringstream sin(lbl);
598  int paneId;
599  sin >> paneId;
600 
601  // Create and initialize a Block object.
602  block = new Block_HDF4(pathv[i], geomFile, geomIdx, paneId, timeLevel,
603  units, numPoints, ghostPoints);
604 
605  // Add grid info.
606  if (gridType == 2) {
607  // Structured grids only have dimensions.
608  block->m_gridInfo.push_back(GridInfo_HDF4(size, ghostCells[0]));
609  } else {
610  // Get info for each connectivity table. If we were given ghost
611  // cell info, then we know how many connectivity tables we have.
612  // Otherwise we have to keep going until we find a dataset that is
613  // obviously not a connectivity table.
614  std::vector<int>::iterator p;
615  for (p=ghostCells.begin();
616  ghostCells.empty() || p!=ghostCells.end();
617  ++p,++index3) {
618  sds_id = HDF4::Select(geom_id, index3, name, &rank, size,
619  &dType, &nAttrs);
620  if (sds_id == FAIL) {
621  if (!ghostCells.empty())
622  std::cerr << "Rocstar: Warning: Select() failed on connectivity dataset "
623  << index3 << " in file "
624  << (geomFile.empty() ? pathv[i] : geomFile.c_str())
625  << std::endl;
626  break;
627  }
628  AutoCloser<int32, intn> auto3(sds_id, HDF4::SDendaccess);
629 
630  // If the rank isn't 2, then it's not a connectivity table.
631  if (rank != 2) {
632  break;
633  }
634 
635  // Get metadata on the connectivity table.
636  HDF4_CHECK_RET(HDF4::SDgetdatastrs,
637  (sds_id, label, units, varName, varType, MAX_NC_NAME),
638  continue);
639 
640  int ghostCount = 0;
641  if (!ghostCells.empty())
642  ghostCount = *p;
643 
644  // A connectivity table must start with ':'
645  if (varName[0] != ':') { /* Old, pre-roccom3 dataset */
646  /* HIDEOUS HACK: should figure out real element type */
647  if (gridType == 3 || gridType == 4) {
648  if (size[0] == 4)
649  strcpy(varName,":T4:");
650  else if (size[0] == 5)
651  strcpy(varName,":P5:");
652  else if (size[0] == 6)
653  strcpy(varName,":P6:");
654  else if (size[0] == 8)
655  strcpy(varName,":B8:");
656  else if (size[0] == 10)
657  strcpy(varName,":T10:");
658  else if (size[0] == 20)
659  strcpy(varName,":B20:");
660  else
661  strcpy(varName,":unknown");
662  } else if (gridType == 5) {
663  if (size[0] == 3)
664  strcpy(varName,":t3:");
665  else if (size[0] == 4)
666  strcpy(varName,":q4:");
667  else if (size[0] == 6)
668  strcpy(varName,":t6:");
669  else if (size[0] == 8)
670  strcpy(varName,":q8:");
671  else
672  strcpy(varName,":unknown");
673  } else {
674  strcpy(varName,":unknown");
675  }
676 
677  if (1) {
678  std::cerr << "Rocstar: Warning: reading old-style connectivity dataset "
679  << varName << " from " << index3 << " in file "
680  << (geomFile.empty() ? pathv[i] : geomFile.c_str())
681  << std::endl;
682  }
683  } else {
684  // If the table is empty, set number of elements to 0.
685  char* pos = strchr(label, '|');
686  if (pos && *(pos+2)=='0') {
687  size[1] = ghostCount = 0;
688  } else {
689  // Get a trustworthy count of the number of ghost elements.
690  std::istringstream in(pos+3);
691  in >> ghostCount;
692  }
693  }
694 
695  if (size[1]<0 || ghostCount < 0) {
696  std::cerr << "Rocstar: Error: Bad number of element or ghosts for attribute " << varName << " nElem=" << size[1] << " nGhost=" << ghostCount << std::endl;
697  COM_assertion_msg(0,
698  "Bad number of element or ghosts for attribute.");
699  }
700  // Store the mesh info.
701  block->m_gridInfo.push_back(GridInfo_HDF4(varName, size[1], ghostCount, index3));
702  }
703  }
704  if (geom_id != sd_id) {
705  HDF4::SDend(geom_id);
706  } else {
707  index = index3;
708  }
709 
710  // Now search for datasets with variable data.
711  for (++index; index<nDataSets; ++index) {
712  HDF4_CHECK_RET(sds_id = HDF4::Select,
713  (sd_id, index, name, &rank, size, &dType, &nAttrs, nDataSets),
714  continue);
715  AutoCloser<int32, intn> auto4(sds_id, HDF4::SDendaccess);
716 
717  // Get metadata on the variable.
718  HDF4_CHECK_RET(HDF4::SDgetdatastrs,
719  (sds_id, label, units, varName, varType, MAX_NC_NAME),
720  continue);
721 
722  if (varName[0] == '\0' || varName[0] == ':') {
723  // Hmm, this is a geometry dataset.
724  continue;
725  }
726 
727  if (strncasecmp(varName, "block header", 12) == 0) {
728  // We've found another block header. We're done.
729  // Reverse index by one, as it will be incremented by the outer-loop
730  --index;
731  break;
732  }
733 
734  // Determine the location. Format of label is (FIXME: complete?)
735  // "<component letter: 1,2,3,x,y,z>-<attribute>"
736  // "|<location letter: w,p,c,n,e>"
737  // " AT TIME <time SS.MMMUUU> "
738  int ng=0; bool is_null=false;

741  char* pos = strchr(label, '|');
742  if (pos == NULL) {

744  guess the data location from the size of the data:
745  */
746  int sz=size[0]; // Should this be size[rank-1]?
747 
748  location='p'; /* guess a panel attribute */
749

751  if (num==sz) {location='n'; ng=0; /* nodes */ }
752  num+=g;

758  num+=g;

764  << varName << "' in dataset " << index << " in file "

767  << "(" << sz << ")" << std::endl;

770  for (int i=0;i<rank;i++) std::cerr<<size[i]<<',';
771  std::cerr << std::endl;
772  }
773  }

775  location = *(pos + 1);
776  // Also obtain the number of ghost items
777  if ( *(pos+2) == '0')
778  { size[0] = ng = 0; is_null=true; }
779  else if ( (*(pos + 2)==',' || *(pos + 2)=='@'))

799  break;

803  nc = 9;
804  break;
805 
806  case 2: case 3: {
807  // Second or third component of vector data.
808  VarInfo_HDF4 &vi = block->m_variables.back();
809  if (vi.m_name != varName || vi.m_indices.size() != 3) {
810  // We shouldn't be seeing 2 or 3 before we see 1.
811  std::cerr << "Rocstar: Warning: Found a vector component named " << varName
812  << " for which there is no X component at "
813  << " dataset " << index << " (" << name
814  << ") in file " << pathv[i] << std::endl;
815  continue;
816  }
817  vi.m_indices[vType-1] = index;
818  vi.m_is_null[vType-1] = is_null;
819  continue;
820  }
821  case 12: case 13:
822  case 21: case 22: case 23:
823  case 31: case 32: case 33: {
824  // Second, third, etc component of tensor data.
825  VarInfo_HDF4 &vi = block->m_variables.back();
826  if (vi.m_name != varName || vi.m_indices.size() != 9) {
827  // We shouldn't be seeing any of these before we see 11.
828  std::cerr << "Rocstar: Warning: Found a tensor component named " << varName
829  << " for which there is no (0,0) component at "
830  << "dataset " << index << " (" << name
831  << ") in file " << pathv[i] << std::endl;
832  continue;
833  }
834  vi.m_indices[3*(vType/10)+(vType%10)-4] = index;
835  vi.m_is_null[3*(vType/10)+(vType%10)-4] = is_null;
836  continue;
837  }
838  default:
839  // I don't know what kind of variable this is supposed to be.
840  // Assuming scalar.
841  vType = 0;
842  nc = 1;
843  break;
844  }
845 
846  int offset = 0; // Starting point for actual attribute name.
847  if ( varName[0]>='0' && varName[0]<='9') {
848  if ( std::strncmp( varName, "1-", 2)==0)
849  offset = 2; // Skip the "1-" part
850  else { // Append to the back
851  VarInfo_HDF4 &vi = block->m_variables.back();
852  unsigned int id = std::atoi( varName);
853  const char *aName = std::strchr(varName, '-');
854 
855  if ( aName==NULL || vi.m_name != (aName+1) ||
856  vi.m_indices.size() != id-1) {
857  // The individual components must be stored consecutively
858  std::cerr << "Rocstar: Warning: Found a vector named " << vi.m_name
859  << " whose components are not stored together "
860  << " in file " << pathv[i] << std::endl;
861  continue;
862  }
863 
864  vi.m_indices.push_back( index);
865  vi.m_is_null.push_back( is_null);
866  DEBUG_MSG("Correction: variable '" << vi.m_name << "' actually has "
867  << vi.m_indices.size() << " components.");
868  continue;
869  }
870  }
871 
872  DEBUG_MSG("Found variable '" << &varName[offset] << "', location == '"
873  << location << "', nComp == " << nc << ", nItems == "
874  << size[0] << ", nGhost == " << ng << ", isNull == "
875  << is_null);
876  block->m_variables.push_back
877  (VarInfo_HDF4(&varName[offset], location, HDF2COM[dType],
878  units, nc, index, size[0], ng, is_null));
879  }
880  blocks.insert(BlockMM_HDF4::value_type(matName, block));
881  }
882  }
