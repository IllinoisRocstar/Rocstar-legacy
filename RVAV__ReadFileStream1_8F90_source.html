43 SUBROUTINE rvav_readfilestream1 ( regionsS1 )
44 
45  USE moddatatypes
46  USE moderror
47  USE moddatastruct, ONLY : t_region
48  USE modglobal, ONLY : t_global
49  USE modgrid, ONLY : t_grid
50  USE modmixture, ONLY : t_mixt
51  USE modinterfaces, ONLY : rflo_readregiontopology, rflo_initinputvalues, &
52  readinputfile, rflo_derivedinputvalues, rflo_getdimensdummynodes, &
53  rflo_getdimensdummy, rflo_getnodeoffset, rflo_getcelloffset, &
54  rflo_readgridregion, rflo_readsolutionregion, mixtureproperties
55  USE modmpi
56  USE modparameters
57  USE rvav_modparameters
58  USE rvav_modglobal
59  USE rvav_moddatastruct
60  IMPLICIT NONE
61 
62 #include "Indexing.h"
63 
64 ! ... parameter
65  TYPE (t_region), POINTER :: regionss1(:)
66 
67 ! ... loop variables
68  INTEGER :: ireg, ilev
69  INTEGER :: i, j, k, ijkn0
70 
71 ! ... local variables
72  CHARACTER(CHRLEN) :: msg
73 
74  INTEGER :: ipc, jpc, kpc, ibc, iec, ibn, ien
75  INTEGER :: idcbeg, idcend, jdcbeg, jdcend, kdcbeg, kdcend
76  INTEGER :: idnbeg, idnend, jdnbeg, jdnend, kdnbeg, kdnend
77  INTEGER :: icoff, ijcoff, inoff, ijnoff, errorflag
78 
79  REAL(RFREAL), DIMENSION(3) :: xyzmin, xyzmax
80  REAL(RFREAL), DIMENSION(5) :: cvmin , cvmax
81 
82  TYPE(t_grid), POINTER :: grid
83  TYPE(t_mixt), POINTER :: mixt
84  TYPE(t_global), POINTER :: global
85 
86 !******************************************************************************
87 
88  global => regionss1(1)%global
89 
90  CALL registerfunction( global, 'RVAV_ReadFileStream1',&
91  'RVAV_ReadFileStream1.F90' )
92 
93 ! load case name
94 
95  globalrvav%casename = trim(global%casename)
96 
97 ! modify all global data pertinent to Rocflo for code reuse
98 
99  global%casename = trim(globalrvav%casename)//'_s1'
100 
101  global%flowType = globalrvav%flowTypeS1
102  global%gridFormat = globalrvav%gridFormatS1
103  global%solutFormat = globalrvav%solutFormatS1
104 
105  IF ( global%verbLevel/=verbose_none ) THEN
106  WRITE(stdout,'(/,A,I5)')'FileTypeS1 = ', globalrvav%fileTypeS1
107  WRITE(stdout,'(A,I5)')'FlowTypeS1 = ', globalrvav%flowTypeS1
108  WRITE(stdout,'(A,I5)')'gridFormatS1 = ', globalrvav%gridFormatS1
109  WRITE(stdout,'(A,I5)')'solutFormatS1 = ', globalrvav%solutFormatS1
110  END IF ! verbLevel
111 
112  IF ( globalrvav%fileTypeS1 == file_computed ) THEN
113 
114 ! loop over regions -----------------------------------------------------------
115 
116  IF ( global%verbLevel/=verbose_none ) &
117  WRITE(stdout,'(/,A)') 'Reading grid and solution from Stream1 ...'
118 
119  DO ireg=1,global%nRegions
120 
121  IF (global%verbLevel /= verbose_none) THEN
122  WRITE(stdout,'(A,I5.5)') ' - region ',ireg
123  WRITE(stdout,'(A,I5.5)') ' global_startLevel ',global%startLevel
124  ENDIF
125 
126  ilev = regionss1(ireg)%currLevel
127  grid => regionss1(ireg)%levels(ilev)%grid
128  mixt => regionss1(ireg)%levels(ilev)%mixt
129 
130 ! --- allocate memory
131 
132  CALL rflo_getdimensdummy( regionss1(ireg),ilev,idcbeg,idcend, &
133  jdcbeg,jdcend,kdcbeg,kdcend )
134  CALL rflo_getcelloffset( regionss1(ireg),ilev,icoff,ijcoff )
135  ibc = indijk(idcbeg,jdcbeg,kdcbeg,icoff,ijcoff)
136  iec = indijk(idcend,jdcend,kdcend,icoff,ijcoff)
137 
138  CALL rflo_getdimensdummynodes( regionss1(ireg),ilev,idnbeg,idnend, &
139  jdnbeg,jdnend,kdnbeg,kdnend )
140  CALL rflo_getnodeoffset( regionss1(ireg),ilev,inoff,ijnoff )
141  ibn = indijk(idnbeg,jdnbeg,kdnbeg,inoff,ijnoff)
142  ien = indijk(idnend,jdnend,kdnend,inoff,ijnoff)
143 
144  globalrvav%iCOffS1 = icoff
145  globalrvav%ijCOffS1 = ijcoff
146 
147  ALLOCATE( grid%xyz(3,ibn:ien),stat=errorflag )
148  IF (regionss1(ireg)%mixtInput%moveGrid) THEN
149  ALLOCATE( grid%siVel(ibn:ien),stat=errorflag )
150  ALLOCATE( grid%sjVel(ibn:ien),stat=errorflag )
151  ALLOCATE( grid%skVel(ibn:ien),stat=errorflag )
152  ENDIF
153  global%error = errorflag
154  IF (global%error /= 0) CALL errorstop( global, err_allocate,__line__ )
155 
156  ALLOCATE( mixt%cv(5,ibc:iec) ,stat=errorflag )
157  global%error = errorflag
158  IF (global%error /= 0) CALL errorstop( global, err_allocate,__line__ )
159 
160  ALLOCATE( mixt%dv(mixt%nDv,ibc:iec),stat=errorflag )
161  global%error = errorflag
162  IF (global%error /= 0) CALL errorstop( global, err_allocate,__line__ )
163 
164  IF (regionss1(ireg)%mixtInput%computeTv) THEN
165  ALLOCATE( mixt%tv(mixt%nTv,ibc:iec),stat=errorflag )
166  global%error = errorflag
167  IF (global%error /= 0) CALL errorstop( global, err_allocate,__line__ )
168  ENDIF
169 
170  IF (regionss1(ireg)%mixtInput%gasModel == gas_model_tcperf) THEN
171  ALLOCATE( mixt%gv(mixt%nGv,0:1),stat=errorflag )
172  global%error = errorflag
173  IF (global%error /= 0) CALL errorstop( global, err_allocate,__line__ )
174  ELSE
175  ALLOCATE( mixt%gv(mixt%nGv,ibc:iec),stat=errorflag )
176  global%error = errorflag
177  IF (global%error /= 0) CALL errorstop( global, err_allocate,__line__ )
178  ENDIF
179 
180 ! --- initialize variables
181 
182  grid%xyz = 0.0_rfreal
183  mixt%cv = 0.0_rfreal
184 
185 ! --- read grid
186 
187  CALL rflo_readgridregion( ireg,regionss1 )
188 
189 ! --- read solution, calc. mixture properties
190 
191  CALL rflo_readsolutionregion( ireg,regionss1 )
192 
193  CALL mixtureproperties( regionss1(ireg),ibc,iec,.true. )
194 
195 ! --- write min-max values
196 
197  IF (global%verbLevel /= verbose_none) THEN
198 
199  WRITE(stdout,'(/,A,2(I10,2X))')'ibn-ien ',ibn,ien
200  WRITE(stdout,'(/,A,3(I8,2X))')'idnbeg,jdnbeg,kdnbeg',idnbeg,jdnbeg,kdnbeg
201  WRITE(stdout,'(/,A,3(I8,2X))')'idnend,jdnend,kdnend',idnend,jdnend,kdnend
202 
203  xyzmin = +1.0e+30_rfreal
204  xyzmax = -1.0e+30_rfreal
205 
206  DO i=ibn, ien
207  xyzmin(1) = min(grid%xyz(1,i), xyzmin(1))
208  xyzmin(2) = min(grid%xyz(2,i), xyzmin(2))
209  xyzmin(3) = min(grid%xyz(3,i), xyzmin(3))
210 
211  xyzmax(1) = max(grid%xyz(1,i), xyzmax(1))
212  xyzmax(2) = max(grid%xyz(2,i), xyzmax(2))
213  xyzmax(3) = max(grid%xyz(3,i), xyzmax(3))
214  ENDDO ! i
215 
216  WRITE(stdout,'(/,A,2(E12.5,2X))') 'Min-Max of X ', xyzmin(1),xyzmax(1)
217  WRITE(stdout,'(A,2(E12.5,2X))') 'Min-Max of Y ', xyzmin(2),xyzmax(2)
218  WRITE(stdout,'(A,2(E12.5,2X))') 'Min-Max of Z ', xyzmin(3),xyzmax(3)
219 
220  DO k=kdnbeg+3,kdnbeg+3
221  DO j=jdnbeg+3,jdnbeg+3
222  DO i=idnbeg,idnend
223  ijkn0 = indijk(i ,j,k,inoff,ijnoff)
224  WRITE(stdout,'(I5,3X,E12.5)') i,grid%xyz(1,ijkn0)
225  ENDDO ! i
226  ENDDO ! j
227  ENDDO ! k
228 
229  WRITE(stdout,'(/,A,2(I10,2X))') 'ibc-iec ',ibc,iec
230  WRITE(stdout,'(A,3(I8,2X))') 'idcbeg,jdcbeg,kdcbeg',idcbeg,jdcbeg,kdcbeg
231  WRITE(stdout,'(A,3(I8,2X))') 'idcend,jdcend,kdcend',idcend,jdcend,kdcend
232  WRITE(stdout,'(/,A,2(I8,2X))') 'iCOffS1,ijCOffS1 ',globalrvav%iCOffS1,globalrvav%ijCOffS1
233 
234  cvmin = +1.0e+30_rfreal
235  cvmax = -1.0e+30_rfreal
236 
237  DO i=ibc,iec
238  cvmin(1) = min(mixt%cv(1,i), cvmin(1))
239  cvmin(2) = min(mixt%cv(2,i), cvmin(2))
240  cvmin(3) = min(mixt%cv(3,i), cvmin(3))
241  cvmin(4) = min(mixt%cv(4,i), cvmin(4))
242  cvmin(5) = min(mixt%cv(5,i), cvmin(5))
243 
244  cvmax(1) = max(mixt%cv(1,i), cvmax(1))
245  cvmax(2) = max(mixt%cv(2,i), cvmax(2))
246  cvmax(3) = max(mixt%cv(3,i), cvmax(3))
247  cvmax(4) = max(mixt%cv(4,i), cvmax(4))
248  cvmax(5) = max(mixt%cv(5,i), cvmax(5))
249  ENDDO ! i
250 
251  WRITE(stdout,'(/,A,2(E12.5,2X))') 'Min-Max of CV(1) ', cvmin(1),cvmax(1)
252  WRITE(stdout,'(A,2(E12.5,2X))') 'Min-Max of CV(2) ', cvmin(2),cvmax(2)
253  WRITE(stdout,'(A,2(E12.5,2X))') 'Min-Max of CV(3) ', cvmin(3),cvmax(3)
254  WRITE(stdout,'(A,2(E12.5,2X))') 'Min-Max of CV(4) ', cvmin(4),cvmax(4)
255  WRITE(stdout,'(A,2(E12.5,2X))') 'Min-Max of CV(5) ', cvmin(5),cvmax(5)
256 
257  ENDIF ! verbLevel
258 
259  ENDDO ! iReg
260 
261  ELSE
262 
263  WRITE(*,'(/,A,/)') 'Current Data Structure for Stream 1 Only supports Computed Results.'
264  stop
265 
266  ENDIF ! fileTypes1
267 
268 ! finalize --------------------------------------------------------------------
269 
270  CALL deregisterfunction( global )
271 
272 END SUBROUTINE rvav_readfilestream1
